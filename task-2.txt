# Створення контейнера
функція створити_контейнер(ім'я, образ):
    виконати_команду("mkdir", ім'я)
    виконати_команду("cp", образ, ім'я)
    виконати_команду("chroot", ім'я, "/sbin/init")

# Видалення контейнера
функція видалити_контейнер(ім'я):
    виконати_команду("killall", "-9", "init", "для", ім'я)
    виконати_команду("rm", "-rf", ім'я)

# Виконання команди всередині контейнера
функція виконати_команду_у_контейнері(ім'я, команда):
    виконати_команду("chroot", ім'я, команда)

# Приклад використання
створити_контейнер("my_container", "base_image")
виконати_команду_у_контейнері("my_container", "apt-get", "install", "-y", "nginx")
виконати_команду_у_контейнері("my_container", "service", "nginx", "start")
видалити_контейнер("my_container")
